Draw tables
Sub DrawTables() Dim tbl As Table For Each tbl In ActiveDocument.Tables With tbl .Style = "Table Grid" .Borders.InsideLineStyle = wdLineStyleSingle .Borders.OutsideLineStyle = wdLineStyleSingle .Borders.InsideLineWidth = wdLineWidth050pt .Borders.OutsideLineWidth = wdLineWidth050pt .AutoFitBehavior (wdAutoFitWindow) End With Next tbl MsgBox "Done! All tables have been formatted." End Sub
Align columns
Sub AlignColumns() Dim tbl As Table Dim r As Row On Error Resume Next Application.ScreenUpdating = False For Each tbl In ActiveDocument.Tables With tbl .AllowAutoFit = False .PreferredWidthType = wdPreferredWidthPoints .PreferredWidth = InchesToPoints(6.27) .Rows.LeftIndent = 0 .Rows.Alignment = wdAlignRowLeft For Each r In .Rows If r.Cells.Count = 3 Then r.Cells(1).Width = InchesToPoints(1.1) r.Cells(3).Width = InchesToPoints(1.1) r.Cells(2).Width = InchesToPoints(4.07) ElseIf r.Cells.Count = 2 Then r.Cells(1).Width = InchesToPoints(1.1) r.Cells(2).Width = InchesToPoints(5.17) End If Next r End With Next tbl Application.ScreenUpdating = True MsgBox "Tables synced to 6.27 inches!" End Sub
Bold answers
Sub BoldCorrectAnswersFromTXT() Dim doc As Document Dim filePath As String Dim fs As Object, ts As Object Dim answerLine As String, parts() As String Dim qNum As Long, correctAns As String Set doc = ActiveDocument Set fs = CreateObject("Scripting.FileSystemObject") With Application.FileDialog(msoFileDialogFilePicker) .Title = "Select your answer key file" .Filters.Add "Text Files", "*.txt", 1 If .Show <> -1 Then Exit Sub filePath = .SelectedItems(1) End With Set ts = fs.OpenTextFile(filePath, 1) Do While Not ts.AtEndOfStream answerLine = Trim(ts.ReadLine) If answerLine <> "" Then parts = Split(answerLine, " ") If UBound(parts) >= 1 Then On Error Resume Next qNum = CLng(parts(0)) correctAns = Trim(parts(1)) On Error GoTo 0 If qNum > 0 And Len(correctAns) > 0 Then Call HighlightSpecificAnswer(doc, qNum, correctAns) End If End If End If Loop ts.Close MsgBox "Processing Complete!", vbInformation End Sub Sub HighlightSpecificAnswer(doc As Document, qNum As Long, ans As String) Dim rng As Range Set rng = doc.Content rng.Collapse Direction:=wdCollapseStart With rng.Find .Text = qNum & ". " .Wrap = wdFindStop If .Execute() Then Dim searchRange As Range Set searchRange = doc.Range(rng.End, doc.Content.End) Set rng = searchRange With rng.Find .Text = "(" & ans & ")" .MatchCase = True .Wrap = wdFindStop If .Execute() Then rng.Expand Unit:=wdSentence Selection.SetRange Start:=rng.Start, End:=rng.End Selection.Expand Unit:=wdParagraph Selection.Font.Bold = True End If End With End If End With End Sub
